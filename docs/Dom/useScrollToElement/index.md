# useScrollToElement

## Introduction

滚动到对应DOM元素节点上

## Basic Usage

```ts
import { useScrollToElement } from '@flypeng/tool/browser';

useScrollToElement(HTMLElement);
```

## Type Declaration

```ts
/**
 * 滚动到对应DOM元素节点上
 * @param element
 */
declare function useScrollToElement(element: HTMLElement): void;
```
